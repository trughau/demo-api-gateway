# ==================================================
# DEMO SCRIPT END-TO-END: API GATEWAY + REDIS + JAEGER
# File này dùng để demo toàn bộ project từ đầu tới cuối.
# ==================================================

# --------------------------------------------------
# 0) Khởi động lại sạch sẽ (khuyến nghị trước khi demo)
# --------------------------------------------------
npm run docker:clean-up

# Nếu muốn chạy thủ công:
# docker-compose down -v
# docker-compose up --build

# Mở Jaeger UI:
# http://localhost:16686

# --------------------------------------------------
# 1) Kiểm tra health hệ thống nhanh
# --------------------------------------------------
curl -i http://localhost:3000

# --------------------------------------------------
# 2) Demo Load Balancing (Round Robin) - User Service
# --------------------------------------------------
# Gọi 2-3 lần, quan sát log gateway:
# [LB] Điều hướng tới Instance: http://user-service:3001
# [LB] Điều hướng tới Instance: http://user-service-2:3001
curl -H "Authorization: Bearer secret-token" http://localhost:3000/api/users
curl -H "Authorization: Bearer secret-token" http://localhost:3000/api/users
curl -H "Authorization: Bearer secret-token" http://localhost:3000/api/users

# --------------------------------------------------
# 3) Demo Redis Cache-Aside cho /api/products
# --------------------------------------------------
# Lần 1: cache miss -> gateway gọi gRPC Product Service -> ghi Redis (setex)

# Lần 2: cache hit -> gateway đọc Redis (get) và trả nhanh hơn
curl -w "\nTime: %{time_total}s\n" -H "Authorization: Bearer secret-token" "http://localhost:3000/api/products?page=1&pageSize=10"

# Gợi ý quan sát Jaeger:
# - Span cha: gateway.listProducts
# - Span con Redis: gateway.redis.get, gateway.redis.setex


# --------------------------------------------------
# 4) Demo Rate Limit (10 request / 10 giây)
# --------------------------------------------------
# Gọi nhanh liên tục >10 lần trong 10 giây, sẽ nhận 429
curl -H "Authorization: Bearer secret-token" http://localhost:3000/api/users

# --------------------------------------------------
# 5) Demo luồng chính /api/checkout (Scatter-Gather Happy Path)
# --------------------------------------------------
# Scatter:
# - Nhánh A: GET user qua REST (có round-robin)
# - Nhánh B: checkStock qua gRPC cho từng cart item
# Gather:
# - Tổng hợp kết quả -> gọi Order Service tạo đơn
curl -X POST http://localhost:3000/api/checkout -H "Authorization: Bearer secret-token" -H "Content-Type: application/json" -d "{\"userId\":\"user-001\",\"cartItems\":[{\"productId\":\"prod-001\",\"quantity\":1}],\"shippingAddress\":\"Hanoi\",\"paymentMethod\":\"COD\"}"

# Kỳ vọng:
# - HTTP 201
# - Jaeger hiển thị flow xanh qua Gateway -> User/Product -> Order


# --------------------------------------------------
# 6) Demo Security - Chặn tại Gateway (x-auth-mode: gateway)
# --------------------------------------------------
# Token sai bị chặn ngay tại gateway.
# Kỳ vọng: HTTP 401, Jaeger thường chỉ thấy span đỏ phía gateway auth check.
curl -X POST http://localhost:3000/api/checkout -H "x-auth-mode: gateway" -H "Authorization: Bearer token-fake" -H "Content-Type: application/json" -d "{\"userId\":\"user-001\",\"cartItems\":[{\"productId\":\"prod-001\",\"quantity\":1}]}"
